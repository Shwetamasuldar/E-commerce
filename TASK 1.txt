<script>
/* =============================================================================
   JAVASCRIPT DATABASE & CORE FUNCTIONALITY - ALL SECTIONS
   ============================================================================= */

// SECTION 1: Mock Database (Products, Users, Orders)
const productsDB = [
    {id:1, name:'iPhone 15 Pro', price:999.99, img:'https://via.placeholder.com/300x200/007bff/ffffff?text=iPhone', desc:'Latest flagship smartphone', stock:10},
    {id:2, name:'MacBook Air M3', price:1299.99, img:'https://via.placeholder.com/300x200/28a745/ffffff?text=MacBook', desc:'Ultra-light laptop', stock:5},
    {id:3, name:'AirPods Pro 2', price:249.99, img:'https://via.placeholder.com/300x200/dc3545/ffffff?text=AirPods', desc:'Noise-cancelling earbuds', stock:20},
    {id:4, name:'Apple Watch Ultra', price:799.99, img:'https://via.placeholder.com/300x200/ffc107/000?text=Watch', desc:'Rugged smartwatch', stock:8},
    {id:5, name:'iPad Pro', price:1099.99, img:'https://via.placeholder.com/300x200/17a2b8/fff?text=iPad', desc:'12.9" professional tablet', stock:12},
    {id:6, name:'Magic Keyboard', price:349.99, img:'https://via.placeholder.com/300x200/6f42c1/fff?text=Keyboard', desc:'Premium wireless keyboard', stock:15}
];

let cart = JSON.parse(localStorage.getItem('cart')) || [];
let users = JSON.parse(localStorage.getItem('users')) || [];
let orders = JSON.parse(localStorage.getItem('orders')) || [];
let currentUser = JSON.parse(localStorage.getItem('currentUser')) || null;

// Initialize app
document.addEventListener('DOMContentLoaded', function() {
    renderProducts();
    updateCartDisplay();
    if (currentUser) showWelcome();
});

// SECTION 1: Product Listing Functions
function renderProducts() {
    const grid = document.getElementById('productsGrid');
    grid.innerHTML = productsDB.map(product => `
        <div class="product-card" onclick="showProductDetail(${product.id})">
            <img src="${product.img}" class="product-img" alt="${product.name}">
            <div class="product-info">
                <h3>${product.name}</h3>
                <div class="price">$${product.price.toFixed(2)}</div>
                <p>${product.desc}</p>
                <button class="btn" onclick="event.stopPropagation(); addToCart(${product.id})">Add to Cart</button>
            </div>
        </div>
    `).join('');
}

function showProductDetail(id) {
    const product = productsDB.find(p => p.id === id);
    document.getElementById('productDetailContent').innerHTML = `
        <div style="display:grid;grid-template-columns:1fr 1fr;gap:30px;">
            <img src="${product.img}" style="width:100%;height:400px;object-fit:cover;border-radius:10px;">
            <div>
                <h1>${product.name}</h1>
                <div class="price" style="font-size:2em;">$${product.price.toFixed(2)}</div>
                <p>${product.desc}</p>
                <p><strong>Stock:</strong> ${product.stock} available</p>
                <button class="btn" onclick="addToCart(${product.id})" style="font-size:1.2em;padding:15px;">🛒 Add to Cart</button>
            </div>
        </div>
    `;
    showSection('productDetail');
}

// SECTION 2: Shopping Cart Functions
function addToCart(productId) {
    const product = productsDB.find(p => p.id === productId);
    const cartItem = cart.find(item => item.id === productId);
    
    if (cartItem) {
        cartItem.quantity += 1;
    } else {
        cart.push({...product, quantity: 1});
    }
    
    localStorage.setItem('cart', JSON.stringify(cart));
    updateCartDisplay();
    alert(`${product.name} added to cart!`);
}

function updateCartDisplay() {
    const totalItems = cart.reduce((sum, item) => sum + item.quantity, 0);
    const totalPrice = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
    
    document.getElementById('cartCount').textContent = totalItems;
    document.getElementById('cartTotal').textContent = totalPrice.toFixed(2);
}

function renderCart() {
    if (cart.length === 0) {
        document.getElementById('cartTable').innerHTML = '<p>Your cart is empty</p>';
        return;
    }
    
    document.getElementById('cartTable').innerHTML = `
        <table>
            <tr><th>Product</th><th>Price</th><th>Qty</th><th>Total</th><th></th></tr>
            ${cart.map(item => `
                <tr>
                    <td>${item.name}</td>
                    <td>$${item.price.toFixed(2)}</td>
                    <td>
                        <input type="number" value="${item.quantity}" min="1" 
                               onchange="updateCartItem(${item.id}, this.value)" style="width:60px;">
                    </td>
                    <td>$${(item.price * item.quantity).toFixed(2)}</td>
                    <td><button class="btn btn-danger" onclick="removeFromCart(${item.id})">Remove</button></td>
                </tr>
            `).join('')}
            <tr style="font-weight:bold;"><td colspan="3">TOTAL</td><td>$${cart.reduce((sum, item) => sum + (item.price * item.quantity), 0).toFixed(2)}</td><td></td></tr>
        </table>
    `;
}

function updateCartItem(id, quantity) {
    const item = cart.find(item => item.id == id);
    if (item) {
        item.quantity = parseInt(quantity);
        if (item.quantity <= 0) {
            removeFromCart(id);
        } else {
            localStorage.setItem('cart', JSON.stringify(cart));
            updateCartDisplay();
            renderCart();
        }
    }
}

function removeFromCart(id) {
    cart = cart.filter(item => item.id != id);
    localStorage.setItem('cart', JSON.stringify(cart));
    updateCartDisplay();
    renderCart();
}

// SECTION 3: User Authentication (Client-side)
function registerUser() {
    const email = document.getElementById('userEmail').value;
    const password = document.getElementById('userPassword').value;
    
    if (users.find(u => u.email === email)) {
        document.getElementById('authMessage').innerHTML = '<span style="color:red;">User already exists!</span>';
        return;
    }
    
    const user = { email, password, orders: [] };
    users.push(user);
    localStorage.setItem('users', JSON.stringify(users));
    document.getElementById('authMessage').innerHTML = '<span style="color:green;">Account created successfully!</span>';
}

function loginUser() {
    const email = document.getElementById('userEmail').value;
    const password = document.getElementById('userPassword').value;
    const user = users.find(u => u.email === email && u.password === password);
    
    if (user) {
        currentUser = user;
        localStorage.setItem('currentUser', JSON.stringify(currentUser));
        showWelcome();
        showSection('products');
    } else {
        document.getElementById('authMessage').innerHTML = '<span style="color:red;">Invalid credentials!</span>';
    }
}

function showWelcome() {
    document.querySelector('.cart-panel').innerHTML += `<br>Welcome, ${currentUser.email}!`;
}

// SECTION 4: Order Processing
function processOrder() {
    if (cart.length === 0) {
        alert('Cart is empty!');
        return;
    }
    
    const name = document.getElementById('customerName').value;
    const address = document.getElementById('customerAddress').value;
    
    if (!name || !address) {
        alert('Please fill in all fields!');
        return;
    }
    
    const order = {
        id: Date.now(),
        items: [...cart],
        total: cart.reduce((sum, item) => sum + (item.price * item.quantity), 0),
        customer: { name, address },
        date: new Date().toLocaleString(),
        status: 'pending'
    };
    
    if (currentUser) {
        currentUser.orders.push(order);
        localStorage.setItem('users', JSON.stringify(users));
    }
    orders.push(order);
    localStorage.setItem('orders', JSON.stringify(orders));
    
    cart = [];
    localStorage.setItem('cart', JSON.stringify(cart));
    
    document.getElementById('orderId').textContent = order.id;
    showSection('orderSuccess');
}

// SECTION 5: Navigation & Section Management
function showSection(sectionName) {
    document.querySelectorAll('.section').forEach(s => s.classList.add('hidden'));
    document.getElementById(sectionName + 'Section').classList.remove('hidden');
    
    if (sectionName === 'cart') renderCart();
    if (sectionName === 'checkout') renderOrderSummary();
}

function renderOrderSummary() {
    document.getElementById('orderSummary').innerHTML = cart.map(item => 
        `<p>${item.name} x${item.quantity} - $${(item.price * item.quantity).toFixed(2)}</p>`
    ).join('') + `<p><strong>Total: $${cart.reduce((sum, item) => sum + (item.price * item.quantity), 0).toFixed(2)}</strong></p>`;
}

function showProductDetail(id) { showSection('productDetail'); }
function showCart() { showSection('cart'); }
function showCheckout() { showSection('checkout'); }
function showAuth() { showSection('auth'); }
function closeProductDetail() { showSection('products'); }
</script>
